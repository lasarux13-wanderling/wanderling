function saveStorage() {
try {
localStorage.setItem(STORAGE_KEY, JSON.stringify({
users: state.users,
currentUser: state.currentUser,
}));
} catch(e) { console.error('Storage save failed', e); }
}

function loadStorage() {
try {
const raw = localStorage.getItem(STORAGE_KEY);
if (!raw) return;
const data = JSON.parse(raw);
if (data.users) state.users = data.users;
if (data.currentUser) state.currentUser = data.currentUser;
} catch(e) { console.error('Storage load failed', e); }
}

function getUser() {
if (!state.currentUser) return null;
return state.users[state.currentUser] || null;
}

function getTrips() {
const u = getUser();
if (!u) return [];
if (!u.trips) u.trips = [];
return u.trips;
}

function handleAuth() {

saveStorage();
showGlobe();
}
}

function formatDate(dateStr) {
if (!dateStr) return '';
try {
const d = new Date(dateStr + 'T12:00:00');
return d.toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
} catch { return dateStr; }
}

// ══════════════════════════════════════════════════════════
// ITINERARY PARSER
// ══════════════════════════════════════════════════════════
const MONTH_MAP = { jan:0,feb:1,mar:2,apr:3,may:4,jun:5,jul:6,aug:7,sep:8,oct:9,nov:10,dec:11,
january:0,february:1,march:2,april:3,june:5,july:6,august:7,september:8,october:9,november:10,december:11 };

// Lines that are definitely NOT port stops
const SKIP_LINE_PATTERNS = [
/\bat\s+sea\b/i,
/\bsea\s+day\b/i,
/\bcruising\b/i,
/\bsailing\b/i,
/\breservation\b/i,
/\bcabin\b/i,
/\bstateroom\b/i,
/\bbooking\b/i,
/\bconfirmation\b/i,
/\bpassenger\b/i,
/\baboard\b/i,
/\bonboard\b/i,
/\bembarkation\s+day\b/i,
/\bdisembarkation\s+day\b/i,
/\bship\s+name\b/i,
/\bvoyage\s+no\b/i,
/\bguest\s+name\b/i,
/\bnight\s+cruise\b/i, // catches "7-Night cruise on..."
/\bcruise\s+on\b/i,
/\bcruise\s+line\b/i,
/\bdeparting\s+from\b/i,
/\bsubtotal\b/i,
/\btotal\b/i,
];

// Column headers that appear in cruise itinerary tables
const HEADER_LINE_RE = /^(day|date|port|activity|arrival|departure|time|description|notes?)\s*$/i;
// Combo header like "Day Date Port" or "Day Port Date"
const COMBO_HEADER_RE = /^(day\s+date|date\s+port|day\s+port|port\s+date|day\s+date\s+port)\s*$/i;

// Day-of-week prefix: "Sun, ", "Monday – ", etc.
const DAY_PREFIX_RE = /^(sun|mon|tue|wed|thu|fri|sat)\w*[,\.\-–—\s]+/i;

// Time range suffix: "– 12:00 PM to 8:00 PM", "– Depart 3:00 PM", "– Arrive 6:00 AM", "12:00 PM - 8:00 PM"
const TIME_SUFFIX_RE = /[\-–—]\s*(depart|arrive|arrival|departure|dock|port)?\s*\d{1,2}:\d{2}\s*(am|pm)?.*/i;
const TIME_RANGE_RE = /\s*\d{1,2}:\d{2}\s*(am|pm)\s*(to|-)\s*\d{1,2}:\d{2}\s*(am|pm).*/i;

// "Return to X" → strip "Return to"
const RETURN_PREFIX_RE = /^return\s+to\s+/i;

function parseItinerary() {
const raw = document.getElementById('itineraryText').value;
if (!raw.trim()) { showToast('Paste some itinerary text first'); return; }

const lines = raw.split('\n').map(l => l.trim()).filter(l => l);
const stops = [];
let guessedName = '';

// ── Detect trip name ──────────────────────────────────────
// Look for a line that reads like a trip title (has "Night" or ship keywords, or is long and descriptive)
const TITLE_RE = /\b(\d+[\-\s]?night|\bcruise\b|\bvoyage\b|\bjourney\b|\bitinerary\b)/i;
for (const line of lines) {
if (line.length > 8 && (TITLE_RE.test(line) || (line.length > 20 && line.length < 120))) {
if (!SKIP_LINE_PATTERNS.some(p => p.test(line)) || TITLE_RE.test(line)) {
guessedName = line;
break;
}
}
}

// ── Process each line ─────────────────────────────────────
for (const line of lines) {
// Skip known junk lines
if (SKIP_LINE_PATTERNS.some(p => p.test(line))) continue;
if (HEADER_LINE_RE.test(line.trim())) continue;
if (COMBO_HEADER_RE.test(line.trim())) continue;
if (line.length < 3) continue;
if (/^\d+$/.test(line)) continue; // bare numbers
if (/^[\d\s\-\/\|]+$/.test(line)) continue; // only digits/separators

// Skip long marketing/description sentences (likely trip title or body text, not a port)
// These have lots of words and look like prose, not port entries
const wordCount = line.split(/\s+/).length;
if (wordCount > 12 && !DAY_PREFIX_RE.test(line)) continue;

// ── Extract date ──────────────────────────────────────
let date = null;
const datePatterns = [
/\b(\d{1,2})[\/\-](\d{1,2})[\/\-](\d{2,4})\b/,
/\b(\w{3,9})\s+(\d{1,2}),?\s+(\d{4})\b/i,
/\b(\d{1,2})\s+(\w{3,9})\s+(\d{4})\b/i,
];
for (const pat of datePatterns) {
const m = line.match(pat);
if (m) { date = parseDate(m, line); if (date) break; }
}

// ── Clean port name ───────────────────────────────────
let portName = line;

// Strip day-of-week prefix (e.g. "Sun, " or "Tuesday – ")
portName = portName.replace(DAY_PREFIX_RE, '');

// Strip time suffixes FIRST (before stripping dates, so "– 12:00 PM" doesn't confuse date parser)
portName = portName.replace(TIME_SUFFIX_RE, '');
portName = portName.replace(TIME_RANGE_RE, '');

// Strip dates
portName = portName
.replace(/\b\d{1,2}[\/\-]\d{1,2}[\/\-]\d{2,4}\b/g, '')
.replace(/\b(jan|feb|mar|apr|may|jun|jul|aug|sep|oct|nov|dec)\w*\s+\d{1,2},?\s*\d{0,4}/gi, '')
.replace(/\b\d{1,2}\s+(jan|feb|mar|apr|may|jun|jul|aug|sep|oct|nov|dec)\w*\s*\d{0,4}/gi, '');

// Strip tab/pipe separators
portName = portName.replace(/[\|\t]+.*$/, '').trim();

// Strip "Return to" prefix
portName = portName.replace(RETURN_PREFIX_RE, '');

// Strip trailing fluff: "– Depart", "– Arrive", "Cruise Port", country suffixes after comma only if long
// But keep "Cape Liberty, NJ" style — only strip if what follows the comma is clearly not part of a port name
// Strategy: take text up to an em-dash or " – " as the port segment
const dashSplit = portName.split(/\s+[–—]\s+/);
if (dashSplit.length > 1) portName = dashSplit[0];

// Clean leading/trailing punctuation and whitespace
portName = portName.replace(/^[\-\,\.\:\s]+|[\-\,\.\:\s]+$/g, '').trim();

// Must have at least 2 letters
if (!portName || portName.length < 2) continue;
if (!/[a-zA-Z]{2,}/.test(portName)) continue;

// Skip if it's still a table-header-like token
if (HEADER_LINE_RE.test(portName)) continue;
if (COMBO_HEADER_RE.test(portName)) continue;

// Skip if it looks like a sentence (many words, no date)
if (portName.split(/\s+/).length > 7 && !date) continue;

// ── Deduplicate adjacent same port ───────────────────
const lastStop = stops[stops.length - 1];
if (lastStop && lastStop.name.toLowerCase() === portName.toLowerCase()) {
if (date && !lastStop.date) lastStop.date = date;
continue;
}

const coords = resolveCoords(portName);
stops.push({ name: portName, date: date || '', coords });
}

// ── Render results ────────────────────────────────────────
const resultEl = document.getElementById('parseResult');
const stopsEl = document.getElementById('parsedStops');
const warnEl = document.getElementById('parseWarning');

if (!stops.length) {
showToast("Couldn't detect any stops — try the Manual tab");
return;
}

state.parsedStops = stops;
const nameInput = document.getElementById('parsedTripName');
if (!nameInput.value) nameInput.value = guessedName.slice(0, 60) || 'My Journey';

// ── Auto-detect trip details from full text ───────────────
const fullText = raw;

// Booking / Reservation number — look for patterns like "Booking: ABC123", "Reservation #: 12345678", "Confirmation: XYZ"
const bookingMatch = fullText.match(/\b(?:booking|reservation|confirmation|booking\s*#|res\.?\s*#?|ref(?:erence)?)\s*[:#\s]*([A-Z0-9]{5,12})\b/i);
if (bookingMatch) {
const el = document.getElementById('parsedBookingNum');
if (!el.value) el.value = bookingMatch[1].toUpperCase();
}

// Stateroom / Cabin — "Stateroom: 9254", "Cabin: 9254", "Room: 9254"
const stateroomMatch = fullText.match(/\b(?:stateroom|cabin|room|suite)\s*[:#\s]*([A-Z]?\d{3,5}[A-Z]?)\b/i);
if (stateroomMatch) {
const el = document.getElementById('parsedStateroomNum');
if (!el.value) el.value = stateroomMatch[1];
}

// Ship name — "aboard the X", "onboard X", "ms X", "mv X", "ship: X", or from title line
const shipPatterns = [
/\b(?:aboard\s+the|onboard\s+the|sailing\s+on\s+the|ship\s*[:#]\s*)([A-Z][A-Za-z\s]{3,30}?)(?:\s*[\n,]|$)/im,
/\b(?:ms|mv|msc|ss|rms|rms)\s+([A-Z][A-Za-z\s]{2,25}?)(?:\s*[\n,]|$)/im,
/cruise\s+on\s+(?:the\s+)?([A-Z][A-Za-z\s]{3,30}?)(?:\s*[\n,]|$)/im,
];
for (const pat of shipPatterns) {
const m = fullText.match(pat);
if (m) {
const el = document.getElementById('parsedShipName');
if (!el.value) el.value = m[1].trim();
break;
}
}

// Cruise line — common names scan
const CRUISE_LINES = ['Royal Caribbean','Celebrity Cruises','Norwegian Cruise Line','NCL','Carnival','Princess Cruises','Holland America','MSC Cruises','Disney Cruise','Viking','Regent','Silversea','Oceania','Azamara','Cunard','P&O','Costa','AIDA'];
for (const line of CRUISE_LINES) {
if (fullText.toLowerCase().includes(line.toLowerCase())) {
const el = document.getElementById('parsedCruiseLine');
if (!el.value) el.value = line;
break;
}
}

stopsEl.innerHTML = stops.map((s, i) => `
<div class="parse-stop-row">
<div class="parse-stop-num">${i + 1}</div>
<div class="parse-stop-num">${i === 0 ? '🔴' : i === stops.length-1 ? '🟢' : '🟡'}</div>
<div class="parse-stop-name">${escHtml(s.name)}</div>
<div class="parse-stop-date">${s.date ? formatDate(s.date) : '—'}</div>
</div>
`).join('');

const unmapped = stops.filter(s => !s.coords);
if (unmapped.length) {
warnEl.innerHTML = `<div class="parse-warning">⚠️ ${unmapped.length} stop(s) couldn't be located on the map: ${unmapped.map(s=>s.name).join(', ')}. They'll be shown as text only.</div>`;
} else {
warnEl.innerHTML = '';
}

resultEl.style.display = 'block';
}

function parseDate(match, line) {
try {
const full = match[0];
// MM/DD/YYYY or DD/MM/YYYY
if (/^\d/.test(full)) {
const parts = full.split(/[\/\-]/);
if (parts.length === 3) {
let y = parseInt(parts[2]);
if (y < 100) y += 2000;
const m = parseInt(parts[0]) - 1;
const d = parseInt(parts[1]);
const dt = new Date(y, m, d);
if (!isNaN(dt)) return dt.toISOString().slice(0,10);
}
}
// Month name
const monthMatch = full.match(/(\w+)/);
if (monthMatch) {
const mKey = monthMatch[1].toLowerCase().slice(0,3);
if (MONTH_MAP[mKey] !== undefined) {
const dayMatch = full.match(/\d+/);
if (dayMatch) {
const yearMatch = full.match(/\d{4}/);
const year = yearMatch ? parseInt(yearMatch[0]) : new Date().getFullYear();
const dt = new Date(year, MONTH_MAP[mKey], parseInt(dayMatch[0]));
if (!isNaN(dt)) return dt.toISOString().slice(0,10);
}
}
}
} catch {}
return null;
}

function resolveCoords(name) {
const key = name.toLowerCase().trim()
.replace(/\bst\.\s/g, 'st ')
.replace(/,.*$/, '') // trim country suffix first for lookup
.trim();
// Direct match
if (PORT_COORDS[key]) return PORT_COORDS[key];

// Try without ", Country"
const bare = name.toLowerCase().replace(/,.*/, '').trim();
if (PORT_COORDS[bare]) return PORT_COORDS[bare];

// Partial match
for (const [k, v] of Object.entries(PORT_COORDS)) {
if (bare.includes(k) || k.includes(bare)) return v;
}

return null;
}

let globeState = {

async function initGlobe() {
const canvas = document.getElementById('globeCanvas');
const rect = canvas.parentElement.getBoundingClientRect();
const topbar = document.querySelector('.topbar');
const bottomNav = document.querySelector('.bottom-nav');
const availH = rect.height - topbar.offsetHeight - bottomNav.offsetHeight;
globeState.width = rect.width;
globeState.height = availH;
canvas.width = rect.width * window.devicePixelRatio;
canvas.height = availH * window.devicePixelRatio;
canvas.style.width = rect.width + 'px';
canvas.style.height = availH + 'px';

globeState.ctx = canvas.getContext('2d');
globeState.ctx.scale(window.devicePixelRatio, window.devicePixelRatio);
globeState.scale = Math.min(rect.width, availH) * 0.44;

globeState.projection = d3.geoOrthographic()
.scale(globeState.scale)
.translate([rect.width / 2, availH / 2])
.clipAngle(90)
.rotate(globeState.rotation);

globeState.path = d3.geoPath(globeState.projection, globeState.ctx);

// Load world data
if (!globeState.world) {
try {
const resp = await fetch('https://cdn.jsdelivr.net/npm/world-atlas@2/countries-110m.json');
globeState.world = await resp.json();
} catch(e) {
console.error('Failed to load world data', e);
}
}

attachGlobeEvents(canvas);
drawGlobe();
}

function drawGlobe() {
const { ctx, projection, path, world, width, height, scale } = globeState;
if (!ctx || !world) return;

ctx.clearRect(0, 0, width, height);

// Ocean
ctx.beginPath();
path({ type: 'Sphere' });
const oceanGrad = ctx.createRadialGradient(width*0.4, height*0.35, 0, width/2, height/2, scale);
oceanGrad.addColorStop(0, '#1a3a6a');
oceanGrad.addColorStop(1, '#0d1e3a');
ctx.fillStyle = oceanGrad;
ctx.fill();

// Graticule
ctx.beginPath();
path(d3.geoGraticule()());
ctx.strokeStyle = 'rgba(255,255,255,0.04)';
ctx.lineWidth = 0.5;
ctx.stroke();

// Countries
const countries = topojson.feature(world, world.objects.countries);
ctx.beginPath();
path(countries);
ctx.fillStyle = '#1e3a2f';
ctx.fill();

// Country borders
ctx.beginPath();
path(topojson.mesh(world, world.objects.countries, (a, b) => a !== b));
ctx.strokeStyle = 'rgba(100,200,120,0.15)';
ctx.lineWidth = 0.5;
ctx.stroke();

// Globe outline
ctx.beginPath();
path({ type: 'Sphere' });
ctx.strokeStyle = 'rgba(74,143,232,0.3)';
ctx.lineWidth = 1.5;
ctx.stroke();

// Draw trips
const trips = getTrips();
trips.forEach(trip => {
if (trip.visible === false) return;
drawTrip(trip);
});
}

function drawTrip(trip) {
const { ctx, projection } = globeState;
const validStops = trip.stops.filter(s => s.coords);
if (validStops.length < 1) return;

// Draw arcs between stops
for (let i = 0; i < validStops.length - 1; i++) {
const from = validStops[i];
const to = validStops[i + 1];
drawDottedArc(from.coords, to.coords, trip.color);
}

// Draw stop dots
validStops.forEach((stop, i) => {
const [lon, lat] = stop.coords;
const projected = projection([lon, lat]);
if (!projected) return;

// Check if visible (front of globe)
const [rx, ry] = globeState.projection.rotate();
const lonRad = (lon + rx) * Math.PI / 180;
const latRad = lat * Math.PI / 180;
if (Math.cos(latRad) * Math.cos(lonRad) < 0) return;

const [px, py] = projected;
const radius = i === 0 || i === validStops.length - 1 ? 6 : 5;
const color = i === 0 ? '#e05252' : i === validStops.length - 1 ? '#4caf84' : '#c9a84c';

ctx.beginPath();
ctx.arc(px, py, radius + 2, 0, Math.PI * 2);
ctx.fillStyle = 'rgba(0,0,0,0.4)';
ctx.fill();

ctx.beginPath();
ctx.arc(px, py, radius, 0, Math.PI * 2);
ctx.fillStyle = color;
ctx.fill();
ctx.strokeStyle = 'rgba(255,255,255,0.8)';
ctx.lineWidth = 1.5;
ctx.stroke();
});
}

function drawDottedArc(from, to, color) {
const { ctx, projection } = globeState;
const [lon1, lat1] = from;
const [lon2, lat2] = to;

const steps = 80;
const geoInterp = d3.geoInterpolate([lon1, lat1], [lon2, lat2]);

let lastVisible = null;
const hexColor = color || '#4a8fe8';
const r = parseInt(hexColor.slice(1,3), 16);
const g = parseInt(hexColor.slice(3,5), 16);
const b = parseInt(hexColor.slice(5,7), 16);

for (let i = 0; i <= steps; i++) {
const t = i / steps;
const [lon, lat] = geoInterp(t);
const pt = projection([lon, lat]);

// Visibility check
const rot = globeState.projection.rotate();
const lonRad = (lon + rot[0]) * Math.PI / 180;
const latRad = lat * Math.PI / 180;
const visible = Math.cos(latRad) * Math.cos(lonRad) > 0;

if (visible && pt) {
// Dotted: only draw every 4th segment
if (i % 4 < 2) {
if (lastVisible) {
ctx.beginPath();
ctx.moveTo(lastVisible[0], lastVisible[1]);
ctx.lineTo(pt[0], pt[1]);
ctx.strokeStyle = `rgba(${r},${g},${b},0.85)`;
ctx.lineWidth = 2;
ctx.setLineDash([]);
ctx.stroke();
}
}
lastVisible = pt;
} else {
lastVisible = null;
}
}
}

function renderGlobeTrips() {
// Re-resolve coords for any stops that might not have them
const trips = getTrips();
trips.forEach(trip => {
trip.stops.forEach(stop => {
if (!stop.coords) stop.coords = resolveCoords(stop.name);
});
});
saveStorage();

if (globeState.ctx) drawGlobe();
}

// ── Globe Interaction ──
function attachGlobeEvents(canvas) {
// Mouse
canvas.addEventListener('mousedown', e => {
globeState.isDragging = true;
globeState.lastPos = [e.clientX, e.clientY];
});
window.addEventListener('mousemove', e => {
if (!globeState.isDragging) return;
const dx = e.clientX - globeState.lastPos[0];
const dy = e.clientY - globeState.lastPos[1];
globeState.lastPos = [e.clientX, e.clientY];
const rot = globeState.projection.rotate();
const sensitivity = 0.3;
globeState.projection.rotate([rot[0] + dx * sensitivity, rot[1] - dy * sensitivity, rot[2]]);
drawGlobe();
});
window.addEventListener('mouseup', () => { globeState.isDragging = false; });

// Wheel zoom
canvas.addEventListener('wheel', e => {
e.preventDefault();
const factor = e.deltaY > 0 ? 0.92 : 1.08;
globeState.scale = Math.max(80, Math.min(1200, globeState.scale * factor));
globeState.projection.scale(globeState.scale);
drawGlobe();
}, { passive: false });

// Touch
canvas.addEventListener('touchstart', e => {
e.preventDefault();
if (e.touches.length === 1) {
globeState.isDragging = true;
globeState.lastPos = [e.touches[0].clientX, e.touches[0].clientY];
} else if (e.touches.length === 2) {
globeState.isDragging = false;
globeState.lastPinchDist = pinchDist(e);
}
}, { passive: false });

canvas.addEventListener('touchmove', e => {
e.preventDefault();
if (e.touches.length === 1 && globeState.isDragging) {
const dx = e.touches[0].clientX - globeState.lastPos[0];
const dy = e.touches[0].clientY - globeState.lastPos[1];
globeState.lastPos = [e.touches[0].clientX, e.touches[0].clientY];
const rot = globeState.projection.rotate();
globeState.projection.rotate([rot[0] + dx * 0.35, rot[1] - dy * 0.35, rot[2]]);
drawGlobe();
} else if (e.touches.length === 2) {
const newDist = pinchDist(e);
if (globeState.lastPinchDist) {
const factor = newDist / globeState.lastPinchDist;
globeState.scale = Math.max(80, Math.min(1200, globeState.scale * factor));
globeState.projection.scale(globeState.scale);
drawGlobe();
}
globeState.lastPinchDist = newDist;
}
}, { passive: false });

canvas.addEventListener('touchend', e => {
e.preventDefault();
globeState.isDragging = false;
globeState.lastPinchDist = null;
}, { passive: false });
}

function pinchDist(e) {
const dx = e.touches[0].clientX - e.touches[1].clientX;
const dy = e.touches[0].clientY - e.touches[1].clientY;
return Math.sqrt(dx*dx + dy*dy);
}
